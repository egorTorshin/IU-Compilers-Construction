routine main() is
    var counter: integer is 0;
    
    print("Before dead loop");
    
    while (3 < 1) loop
        print("This will never execute");
        counter := counter + 1;
    end;
    
    while (100 = 50) loop
        print("This is also unreachable");
        counter := counter + 10;
    end;
    
    print("After dead loops");
    print("Counter: " + counter);
end;
